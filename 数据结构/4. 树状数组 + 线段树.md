```c++
struct BIT {
    int n;
    std::vector<int> tr;
    BIT(int _n) {
        n = _n;
        tr.assign(n + 1, 0);
    }
    void add(int x, int k) {
        for (int i = x; i <= n; i += i & -i) {
            tr[i] += k;
        }
    }
    int query(int x) {
        int res = 0;
        for (int i = x; i; i -= i & -i) {
            res += tr[i];
        }
        return res;
    }
};
```

## 动态开点

源自于 [F. Pathwalks](https://codeforces.com/problemset/problem/960/F)  需要维护 $n$ 个树状数组，每个树状数组的范围是 $[1,10^5]$。技巧是将树状数组的数组换为 `std::map<int, int>`。

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int, int>
#define endl "\n"
/**********************  Core code begins  **********************/

const int MAXN = 1e5 + 7;
// map 实现的 “动态开点树状数组”
struct BIT {
	map<int, int> tr;
	void modify(int x, int k) {
		for (int i = x; i < MAXN; i += i & -i) {
			tr[i] = max(tr[i], k);
		}
	}
	int query(int x) {
		int res = 0;
		for (int i = x; i; i -= i & -i) {
			res = max(res, tr[i]);
		}
		return res;
	}
};

void SolveTest() {
	int n, m;
	cin >> n >> m;

	vector<int> dp(n + 1);
	vector<BIT> bit(n + 1);
	int ans = 0;
	for (int i = 1; i <= m; i++) {
		int u, v, w;
		cin >> u >> v >> w;
		w++;
		dp[v] = bit[u].query(w - 1) + 1;
		bit[v].modify(w, dp[v]);
		ans = max(ans, dp[v]);
	}
	cout << ans << endl;
}

/**********************  Core code ends  ***********************/
signed main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int T = 1;
	// cin >> T;
	for (int i = 1; i <= T; i++) {
		SolveTest();
	}
	return 0;
}
```

```cpp
struct BIT_2d {
```

# 线段树

线段树本质上就是在维护一个数组，优点是可以快速地进行区间批量修改和区间查询。所维护的数组可能是一个大小恒定的序列（普通线段树），也可能是一个值域桶（权值线段树），值域过大的时候可以类比哈希的思想（动态开点线段树）。

维护的技巧：

- 将待维护值数学化简，拆成若干个可用线段树维护的值；
- 维护区间前后缀，将左子树后缀和右子树前缀拼成区间性质；
- 批量建线段树。常见的例子：按数位建 $32$ 棵线段树，按字母建 $26$ 棵线段树。

常见思路：

- 离线将所有询问按右端点排序，扫描线维护已经扫过的区间；

- 树上问题经常使用线段树合并。



## 普通线段树

如果不涉及到区间修改，那么就不需要懒标记 `Tag` 类，也不需要 `pushDown()` 操作。

```cpp
template<class Info>
class SegmentTree {
private:
    int n;
    std::vector<Info> info;
    void pushUp(int p) {
        info[p] = info[p * 2] + info[p * 2 + 1];
    }
public:
    /***************** 初始化 *******************/
    // 注意，用于初始化的 init_ 下标从 1 开始！
    template<class T>
    SegmentTree(std::vector<T> init_) {
        n = init_.size() - 1;
        info.assign(n * 4, {});
        std::function<void(int, int, int)> build = [&](int p, int l, int r) {
            if (l == r) {
                info[p] = init_[l];
                return;
            }
            int mid = (l + r) / 2;
            build(p * 2, l, mid);
            build(p * 2 + 1, mid + 1, r);
            pushUp(p);
        };
        build(1, 1, n);
    }    
    /***************** 单点赋值 *****************/
    void indexAssign(int p, int l, int r, int x, const Info &v) {
        if (l == r) {
            info[p] = v;
            return;
        }
        int mid = (l + r) / 2;
        if (x <= mid) {
            indexAssign(p * 2, l, mid, x, v);
        } else {
            indexAssign(p * 2 + 1, mid + 1, r, x, v);
        }
        pushUp(p);
    }
    void indexAssign(int x, const Info &v) {
        indexAssign(1, 1, n, x, v);
    }
    /***************** 单点增加 *****************/
    template<class T>
    void indexInc(int p, int l, int r, int x, const T &v) {
        if (l == r) {
            info[p].inc(v);
            return;
        }
        int mid = (l + r) / 2;
        if (x <= mid) {
            indexInc(p * 2, l, mid, x, v);
        } else {
            indexInc(p * 2 + 1, mid + 1, r, x, v);
        }
        pushUp(p);
    }
    template<class T>
    void indexInc(int x, const T &v) {
        indexInc(1, 1, n, x, v);
    }
    /***************** 单点查询 *****************/
    Info indexQuery(int p, int l, int r, int x) {
        assert(x >= 1 && x <= n);
        if (l == x && r == x) {
            return info[p];
        }
        int mid = (l + r) / 2;
        if (x <= mid) {
            return indexQuery(p * 2, l, mid, x);
        } else {
            return indexQuery(p * 2 + 1, mid + 1, r, x);
        }
    }
    Info indexQuery(int x) {
        return indexQuery(1, 1, n, x);
    }
    /***************** 区间查询 *****************/
    Info rangeQuery(int p, int l, int r, int x, int y) {
        if (l >= x && r <= y) {
            return info[p];
        }
        int mid = (l + r) / 2;
        Info res = {};
        if (x <= mid) {
            res = res + rangeQuery(p * 2, l, mid, x, y);
        }
        if (y > mid) {
            res = res + rangeQuery(p * 2 + 1, mid + 1, r, x, y);
        }
        return res;
    }
    Info rangeQuery(int x, int y) {
        return rangeQuery(1, 1, n, x, y);
    }
};


struct Info {
    Info() {}
    Info(int x) {}
    void inc(int x) {}
    friend Info operator+(const Info &lhs, const Info &rhs) {}
};
```



## 懒标记线段树

如果涉及到区间修改，那么久需要懒标记。

把 `Tag` 和 `Info` 抽象出来，代码量稍大一点，但是有助于处理复杂的区间维护问题。

```cpp
template<class Info, class Tag>
class LazySegmentTree {
private:
    int n;
    std::vector<Info> info;
    std::vector<Tag> tag;
    void pushUp(int p) {
        info[p] = info[p * 2] + info[p * 2 + 1];
    }
    void apply(int p, const Tag &v) {
        info[p].apply(v);
        tag[p].apply(v);
    }
    void pushDown(int p) {
        apply(p * 2, tag[p]);
        apply(p * 2 + 1, tag[p]);
        tag[p] = {};
    }

public:
    // 注意，用于初始化的 init_ 下标从 1 开始！
    template<class T>
    void init(std::vector<T> &init_) {
        n = init_.size() - 1;
        info.assign(n * 4, Info());
        tag.assign(n * 4, Tag());
        std::function<void(int, int, int)> build = [&](int p, int l, int r) {
            if (l == r) {
                info[p] = init_[l];
                return;
            }
            int mid = (l + r) / 2;
            build(p * 2, l, mid);
            build(p * 2 + 1, mid + 1, r);
            pushUp(p);
        };
        build(1, 1, n);
    }

    LazySegmentTree(int n_ = 0) {
        n = n_;
        info.assign(n * 4, Info());
        tag.assign(n * 4, Tag());
    }

    template<class T>
    LazySegmentTree(std::vector<T> &init_) {
        init(init_);
    }
    
    void indexAssign(int p, int l, int r, int x, const Info &v) {
        if (l == r) {
            info[p] = v;
            return;
        }
        pushDown(p);
        int mid = (l + r) / 2;
        if (x <= mid) {
            indexAssign(p * 2, l, mid, x, v);
        } else {
            indexAssign(p * 2 + 1, mid + 1, r, x, v);
        }
        pushUp(p);
    }

    void indexAssign(int x, const Info &v) {
        indexAssign(1, 1, n, x, v);
    }

    Info rangeQuery(int p, int l, int r, int x, int y) {
        if (l >= x && r <= y) {
            return info[p];
        }
        pushDown(p);
        int mid = (l + r) / 2;
        Info res = {};
        if (x <= mid) {
            res = res + rangeQuery(p * 2, l, mid, x, y);
        }
        if (y > mid) {
            res = res + rangeQuery(p * 2 + 1, mid + 1, r, x, y);
        }
        return res;
    }

    Info rangeQuery(int x, int y) {
        return rangeQuery(1, 1, n, x, y);
    }

    void rangeModify(int p, int l, int r, int x, int y, const Tag &v) {
        if (l >= x && r <= y) {
            apply(p, v);
            return;
        }
        pushDown(p);
        int mid = (l + r) / 2;
        if (x <= mid) {
            rangeModify(p * 2, l, mid, x, y, v);
        }
        if (y > mid) {
            rangeModify(p * 2 + 1, mid + 1, r, x, y, v);
        }
        pushUp(p);
    }

    void rangeModify(int x, int y, const Tag &v) {
        rangeModify(1, 1, n, x, y, v);
    }
};

struct Tag {
    Tag() {}    // 一定要注意 Tag 的默认情况（标记被清空时的状态） 
    void apply(Tag t) {}
};

struct Info {
    Info() {}
    Info(int x) {}
    void apply(Tag t) {}
    friend Info operator+(auto lhs, auto rhs) {}
};
```

01线段树：

[P2572 [SCOI2010] 序列操作](https://www.luogu.com.cn/problem/P2572)

维护一个 01 序列，支持：

- 区间置数 0/1；
- 区间 0/1 取反；
- 求区间和；
- 求区间内连续 $1$ 的最大长度

```cpp
struct Tag {
    int setnum = -1, reverse = 0;   // 先翻转再赋值

    void apply(Tag t) {
        // 同一时刻，一定只存在赋值和翻转的操作之一
        assert(!(t.setnum != -1 && t.reverse != 0));
        if (t.setnum != -1) {
            setnum = t.setnum;
            reverse = 0;
        }
        if (t.reverse) {
            if (setnum != -1) {
                setnum ^= 1;
            } else {
                reverse ^= 1;
            }
        }
    }
};

struct Info {
    int sum = 0, len = 1;
    int pre0 = 0, suf0 = 0, cons0 = 0;
    int pre1 = 0, suf1 = 0, cons1 = 0;
    Info() {
        len = 0;
    }
    Info(int x) {
        len = 1;
        sum = pre1 = suf1 = cons1 = x;
        pre0 = suf0 = cons0 = !x;
    }
    void apply(Tag t) {
        assert(!(t.setnum != -1 && t.reverse != 0));
        if (t.setnum != -1) {
            sum = pre1 = suf1 = cons1 = t.setnum * len;
            pre0 = suf0 = cons0 = (1 - t.setnum) * len;
        }
        if (t.reverse) {
            sum = len - sum;
            std::swap(pre0, pre1);
            std::swap(suf0, suf1);
            std::swap(cons0, cons1);
        }
    }
    friend Info operator+(const Info &lhs, const Info &rhs) {
        Info res;
        res.sum = lhs.sum + rhs.sum;
        res.len = lhs.len + rhs.len;
        if (lhs.sum == lhs.len) {
            res.pre1 = lhs.len + rhs.pre1;
        } else {
            res.pre1 = lhs.pre1;
        }
        if (lhs.sum == 0) {
            res.pre0 = lhs.len + rhs.pre0;
        } else {
            res.pre0 = lhs.pre0;
        }
        if (rhs.sum == rhs.len) {
            res.suf1 = rhs.len + lhs.suf1;
        } else {
            res.suf1 = rhs.suf1;
        }
        if (rhs.sum == 0) {
            res.suf0 = rhs.len + lhs.suf0;
        } else {
            res.suf0 = rhs.suf0;
        }
        res.cons1 = std::max({lhs.cons1, rhs.cons1, lhs.suf1 + rhs.pre1});
        res.cons0 = std::max({lhs.cons0, rhs.cons0, lhs.suf0 + rhs.pre0});
        return res;
    }
};
```



### 加法线段树（简化版）

[P3372 【模板】线段树 1](https://www.luogu.com.cn/problem/P3372)

- 区间加法
- 区间求和

```cpp
template<int MAXN>
struct SegTree {
    std::array<int, MAXN> sum, tag;
    void build(int p, int l, int r, std::vector<int> &init_) {
        if (l == r) {
            sum[p] = init_[l];
            return;
        }
        int mid = (l + r) / 2;
        build(p * 2, l, mid, init_);
        build(p * 2 + 1, mid + 1, r, init_);
        pushUp(p);
    }
    void pushUp(int p) {
        sum[p] = sum[p * 2] + sum[p * 2 + 1];
    }
    void pushDown(int p, int l, int r) {
        if (tag[p] == 0) {
            return;
        }
        int mid = (l + r) / 2;
        sum[p * 2] += (mid - l + 1) * tag[p];
        sum[p * 2 + 1] += (r - mid) * tag[p];
        tag[p * 2] += tag[p];
        tag[p * 2 + 1] += tag[p];
        tag[p] = 0;
    }
    int RangeSum(int p, int l, int r, int x, int y) {
        if (l > y || r < x) {
            return 0;
        }
        if (x <= l && r <= y) {
            return sum[p];
        }
        pushDown(p, l, r);
        int mid = (l + r) / 2;
        return RangeSum(p * 2, l, mid, x, y) + RangeSum(p * 2 + 1, mid + 1, r, x, y);
    }
    void RangeAdd(int p, int l, int r, int x, int y, int k) {
        if (l > y || r < x) {
            return;
        }
        if (x <= l && r <= y) {
            sum[p] += (r - l + 1) * k;
            tag[p] += k;
            return;
        }
        pushDown(p, l, r);
        int mid = (l + r) / 2;
        RangeAdd(p * 2, l, mid, x, y, k);
        RangeAdd(p * 2 + 1, mid + 1, r, x, y, k);
        pushUp(p);
    }
};
```



### 01 线段树（简化版）

- 区间赋值 0/1
- 区间求和

```cpp
struct SegmentTree {
	struct Info {
		int sum = 0, tag = -1;
	};
	vector<Info> info;
	SegmentTree(int n): info(n * 4) {}
	void pushup(int p) {
		info[p].sum = info[ls(p)].sum + info[rs(p)].sum;
	}
	void pushdown(int p, int l, int r) {
		if (info[p].tag == -1) {
			return;
		}
		int mid = (r + l) >> 1;
		info[ls(p)] = {(mid - l + 1) * info[p].tag, info[p].tag};
		info[rs(p)] = {(r - mid) * info[p].tag, info[p].tag};
		info[p].tag = -1;
		assert(info[p].sum >= 0);
		assert(info[ls(p)].sum >= 0);
		assert(info[rs(p)].sum >= 0);
	}
	void build(int p, int l, int r, vector<int> &a) {
		if (l == r) {
			info[p].sum = a[l];
			return;
		}
		int mid = (l + r) >> 1;
		build(ls(p), l, mid, a);
		build(rs(p), mid + 1, r, a);
		pushup(p);
	}
	void modify(int p, int l, int r, int x, int y, int k) {
		if (x > y) {	// 这个判断不能忘，否则会 RE
			return;
		}
		if (x <= l && r <= y) {
			info[p].sum = (r - l + 1) * k;
			info[p].tag = k;
			return;
		}
		int mid = (l + r) >> 1;
		pushdown(p, l, r);
		if (x <= mid) {
			modify(ls(p), l, mid, x, y, k);
		}
		if (y > mid) {
			modify(rs(p), mid + 1, r, x, y, k);
		}
		pushup(p);
		assert(info[p].sum >= 0);
	}
	int qsum(int p, int l, int r, int x, int y) {
		if (x <= l && r <= y) {
			return info[p].sum;
		}
		int mid = (l + r) >> 1, ans = 0;
		pushdown(p, l, r);
		if (x <= mid) {
			ans += qsum(ls(p), l, mid, x, y);
		}
		if (y > mid) {
			ans += qsum(rs(p), mid + 1, r, x, y);
		}
		return ans;
	}
};
```

## 动态开点：线段树分裂与合并

注意事项：

- 如果有区间为负数的情形，递归过程中的 `mid` 必须定义为 `int mid = ((l + r) >> 1); `，而不能定义为 `int mid = (l + r) / 2;`
- 空间能开多大就开多大。

[P5494 【模板】线段树分裂](https://www.luogu.com.cn/problem/P5494)

【题意】

给出一个可重集 $a$（编号为 $1$），它支持以下操作：

- `0 p x y`：将可重集 $p$ 中大于等于 $x$ 且小于等于 $y$ 的值移动到一个新的可重集中（新可重集编号为从 $2$ 开始的正整数，是上一次产生的新可重集的编号+1）。
- `1 p t`：将可重集 $t$ 中的数放入可重集 $p$，且清空可重集 $t$（数据保证在此后的操作中不会出现可重集 $t$）。

- `2 p x q`：在 $p$ 这个可重集中加入 $x$ 个数字 $q$。

- `3 p x y`：查询可重集 $p$ 中大于等于 $x$ 且小于等于 $y$ 的值的个数。
- `4 p k`：查询在 $p$ 这个可重集中第 $k$ 小的数，不存在时输出 `-1`。

【题解】

```cpp
#include <bits/stdc++.h>
#define int long long

struct Node {
    int sum = 0;
    Node *ls = nullptr, *rs = nullptr;
};
#define getsum(p) (p ? p->sum : 0)
#define pushUp(p) (p ? p->sum = getsum(p->ls) + getsum(p->rs) : 1)

struct DynamicSegmentTree {
    // 将给定序列建成线段树
    Node *build(int l, int r, std::vector<int> &a) {
        Node *p = new Node;
        if (l == r) {
            p->sum = a[l];
            return p;
        }
        int mid = (l + r) / 2;
        p->ls = build(l, mid, a);
        p->rs = build(mid + 1, r, a);
        pushUp(p);
        return p;
    }

    // 线段树分裂
    Node *split(Node *&p, int l, int r, int x, int y) {
        if (p == nullptr || x > r || y < l) {
            return nullptr;
        } 
        if (l >= x && r <= y) {
            auto res = p;
            p = nullptr;
            return res;
        }
        Node *res = new Node;
        int mid = (l + r) / 2;
        res->ls = split(p->ls, l, mid, x, y);
        res->rs = split(p->rs, mid + 1, r, x, y);
        pushUp(p);
        pushUp(res);
        return res;
    }

    // 合并两棵线段树
    Node *merge(Node *p1, Node *p2, int l, int r) {
        if (p1 == nullptr) {
            return p2;
        }
        if (p2 == nullptr) {
            return p1;
        }
        if (l == r) {
            p1->sum += p2->sum;
            return p1;
        }
        int mid = (l + r) / 2;
        p1->ls = merge(p1->ls, p2->ls, l, mid);
        p1->rs = merge(p1->rs, p2->rs, mid + 1, r);
        pushUp(p1);
        return p1;
    }

    // 单点修改
    Node *indexInc(Node *p, int l, int r, int x, int k) {
        if (p == nullptr) {
            p = new Node;
        }
        if (l == x && r == x) {
            p->sum += k;
            return p;
        }
        int mid = (l + r) / 2;
        if (x <= mid) {
            p->ls = indexInc(p->ls, l, mid, x, k);
        } else {
            p->rs = indexInc(p->rs, mid + 1, r, x, k);
        }
        pushUp(p);
        return p;
    }

    // 查询区间和
    int query(Node *p, int l, int r, int x, int y) {
        if (p == nullptr) {
            return 0;
        }
        if (l >= x && r <= y) {
            return p->sum;
        }
        int mid = (l + r) / 2, res = 0;
        if (x <= mid) {
            res += query(p->ls, l, mid, x, y);
        }
        if (y > mid) {
            res += query(p->rs, mid + 1, r, x, y);
        }
        return res;
    }

    // 查询全局第 k 小
    int k_th(Node *p, int l, int r, int k) {
        if (getsum(p) < k) {
            return -1;
        }
        if (l == r) {
            return l;
        }
        int mid = (l + r) >> 1;
        if (getsum(p->ls) >= k) {
            return k_th(p->ls, l, mid, k);
        } else {
            return k_th(p->rs, mid + 1, r, k - getsum(p->ls));
        }
    }
};

signed main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0);
    std::cout.tie(0);

    int n, m;
    std::cin >> n >> m;
    std::vector<Node*> root(m + 2);
    std::vector<int> a(n + 1);

    for (int i = 1; i <= n; i++) {
        std::cin >> a[i];
    }
    int segcnt = 0;
    DynamicSegmentTree tr;
    root[++segcnt] = tr.build(1, n, a);

    for (int i = 1; i <= m; i++) {
        int op;
        std::cin >> op;
        if (op == 0) {
            int p, x, y;
            std::cin >> p >> x >> y;
            root[++segcnt] = tr.split(root[p], 1, n, x, y);
        }
        if (op == 1) {
            int p, t;
            std::cin >> p >> t;
            tr.merge(root[p], root[t], 1, n);
        }
        if (op == 2) {
            int p, k, x;
            std::cin >> p >> k >> x;
            tr.indexInc(root[p], 1, n, x, k);
        }
        if (op == 3) {
            int p, x, y;
            std::cin >> p >> x >> y;
            std::cout << tr.query(root[p], 1, n, x, y) << "\n";
        }
        if (op == 4) {
            int p, k;
            std::cin >> p >> k;
            std::cout << tr.k_th(root[p], 1, n, k) << "\n";
        }
    }
    return 0;
}
```

```c++
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int, int>
#define endl "\n"
/**********************  Core code begins  **********************/

#define ls(p) (p * 2)
#define rs(p) (p * 2 + 1)

/**
 * @attention 这个线段树当中，对于区间修改操作并没有打懒标记和 pushdown 操作，
 * 原因在于本题保证了所有的正线段都存在对称的负线段。
 * 如果题目不是矩形覆盖，而是单纯地给出若干线段求包围的面积，
 * 那么就不能这样写，需要动态开点老老实实打懒标记和 pushdown 操作。
*/
struct SegmentTree {
	/**
	 * @attenti 线段树的叶子节点不是原本矩形的折点，而是离散化后的一段段区间
	 * @param sum 这个区间被覆盖的长度
	 * @param len 这个区间总共的物理长度
	 * @param cnt 这个区间被完整覆盖的次数
	*/
	struct Info {
		int sum = 0, len = 0, cnt = 0; 
	};

	vector<Info> info;
	SegmentTree(int n) : info(n * 4) {}

	void pushup(int p, int l, int r) {
		if (l == r) {
			info[p].sum = info[p].cnt ? info[p].len : 0;
		} else if (info[p].cnt) {
			info[p].sum = info[ls(p)].len + info[rs(p)].len;
		} else {
			info[p].sum = info[ls(p)].sum + info[rs(p)].sum;
		}
	}

	void build(int p, int l, int r, vector<int> &a) {
		if (l == r) {
			info[p].len = a[l];
			return;
		}
		int mid = (l + r) >> 1;
		build(ls(p), l, mid, a);
		build(rs(p), mid + 1, r, a);
		info[p].len = info[ls(p)].len + info[rs(p)].len;
	}

	void modify(int p, int l, int r, int x, int y, int k) {
		if (l > y || r < x) {
			return;
		}
		if (x <= l && r <= y) {
			info[p].cnt += k;
			pushup(p, l, r);
			return;
		}
		int mid = (l + r) >> 1;
		if (x <= mid) {
			modify(ls(p), l, mid, x, y, k);
		}
		if (y > mid) {
			modify(rs(p), mid + 1, r, x, y, k);			
		}
		pushup(p, l, r);
	}
};

struct Line {
	int x, y1, y2, k;
};

void SolveTest() {
	int n;
	cin >> n;
	vector<Line> line;
	vector<int> tmp;
	for (int i = 1; i <= n; i++) {
		int x1, y1, x2, y2;
		cin >> x1 >> y1 >> x2 >> y2;
		line.push_back({x1, y1, y2, 1});
		line.push_back({x2, y1, y2, -1});
		tmp.push_back(y1);
		tmp.push_back(y2);
	}
	sort(line.begin(), line.end(), [](Line a, Line b) {
		return a.x < b.x;
	});

	sort(tmp.begin(), tmp.end());
	int m = unique(tmp.begin(), tmp.end()) - tmp.begin();
	auto rnk = [&](int x) {
		return lower_bound(tmp.begin(), tmp.begin() + m, x) - tmp.begin() + 1;
	};

	SegmentTree tr(m + 1);
	vector<int> a(m + 1);
	for (int i = 1; i < m; i++) {
		a[i] = tmp[i] - tmp[i - 1];
	}
	m--;
	tr.build(1, 1, m, a);

	int ans = 0;
	for (int i = 0; i < line.size(); i++) {
		if (i > 0) {
			ans += tr.info[1].sum * (line[i].x - line[i - 1].x);
		}
		tr.modify(1, 1, m, rnk(line[i].y1), rnk(line[i].y2) - 1, line[i].k);
	}
	cout << ans << endl;
}

/**********************  Core code ends  ***********************/

	cin.tie(0);
	cout.tie(0);
	int T = 1;
	// cin >> T;
	for (int i = 1; i <= T; i++) {
		SolveTest();
	}
	return 0;
}
```

### 永久化标记

[P3437 [POI2006]TET-Tetris 3D](https://www.luogu.com.cn/problem/P3437)

这个题需要实现：

- 二维矩形区间查询最大值
- 二维矩形区间赋值，保证赋的值不小于该矩形区间最大值

这个问题需要一个技巧：标记永久化。这个技巧使用的前提是所有的修改操作都是非减的。

对于一维的情形，我们维护两个值：
- `mxv[p]` ：记录间内的最大值
-  `tag[p]`：表示区间内的所有值都赋值为 `tag[p]`

对于修改操作，路径上的每一个节点都用 `k` 更新 `mxv[p]`；而只有当前区间被目标区间完全覆盖的时候才更新 `tag[p]`。

对于查询操作，路径上的每一个节点都用 `tag[p]` 更新答案；而只有当前区间被目标区间完全覆盖的时候才用 `mxv[p]` 更新答案。

永久化标记的好处是，不需要进行 `pushup()` 和 `pushdown()` 的操作。所以you局限性。

```cpp
#define lp (p * 2)
#define rp (p * 2 + 1)

struct Segtr {
    int n;
    vector<int> mxv, tag;

    Segtr(int _n) : n(_n) {
        mxv.assign(n * 2 + 100, 0);
        tag = mxv;
    }

    // 区间赋值
    void modify(int p, int l, int r, int x, int y, int k) {
        if (l > y || r < x) {
            return;
        }
        mxv[p] = max(mxv[p], k);
        if (x <= l && r <= y) {
            tag[p] = max(tag[p], k);
            return;
        }
        int mid = (l + r) >> 1;
        modify(lp, l, mid, x, y, k);
        modify(rp, mid + 1, r, x, y, k);
    }

    // 查询区间最大值
    int query(int p, int l, int r, int x, int y) {
        if (l > y || r < x) {
            return -1;
        }
        if (x <= l && r <= y) {
            return mxv[p];
        }
        int mid = (l + r) >> 1;
        return max({ tag[p], query(lp, l, mid, x, y), query(rp, mid + 1, r, x, y) });
    }
};

struct Seg_Seg {
    int n, m;
    vector<Segtr> mxv, tag;

    Seg_Seg(int _n, int _m) : n(_n), m(_m) {
        mxv.assign(n * 2 + 100, Segtr(m));
        tag = mxv;
    }

    // 二维区间赋值
    void modify(int p, int l, int r, int x, int y, int d, int u, int k) {
        if (l > y || r < x) {
            return;
        }
        mxv[p].modify(1, 1, m, d, u, k);

        if (x <= l && r <= y) {
            tag[p].modify(1, 1, m, d, u, k);
            return;
        }
        int mid = (l + r) >> 1;
        modify(lp, l, mid, x, y, d, u, k);
        modify(rp, mid + 1, r, x, y, d, u, k);
    }

    // 查询二维区间最大值
    int query(int p, int l, int r, int x, int y, int d, int u) {
        if (l > y || r < x) {
            return 0;
        }
        if (x <= l && r <= y) {
            return mxv[p].query(1, 1, m, d, u);
        }
        int mid = (l + r) >> 1;
        int ans = tag[p].query(1, 1, m, d, u);
        ans = max({ ans, query(lp, l, mid, x, y, d, u), query(rp, mid + 1, r, x, y, d, u) });
        return ans;
    }
};
```
